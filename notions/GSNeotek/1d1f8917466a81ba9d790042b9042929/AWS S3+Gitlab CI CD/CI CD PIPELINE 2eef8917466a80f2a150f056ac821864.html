# 3-A. Zendesk App Update via ZCLI (Sequential with Variable Selection)
# ==========================================
update_zendesk_dev_apps_zcli:
  stage: update-zendesk
  needs: ["deploy_to_dev"]
  when: manual
  allow_failure: false

  variables:
    # DEPLOY_CLIENTS: 배포할 고객사 선택 (예: "CLIENT1", "CLIENT2", "CLIENT1,CLIENT2")
    # 비워두거나 설정하지 않으면 모든 고객사 배포
    DEPLOY_CLIENTS: ""

  script:
    - |
      set -e

      echo "=== Zendesk App Update via ZCLI (Sequential) START ==="

      # zcli 설치 확인
      if ! command -v zcli &> /dev/null; then
        echo "Installing zcli..."
        npm install -g @zendesk/zcli
      fi

      APP_DIR="zendesk-private-app-dev"
      APP_DIR_ABS="$CI_PROJECT_DIR/$APP_DIR"

      # 클라이언트 목록 결정 (DEPLOY_CLIENTS 변수 또는 기본값)
      if [ -n "$DEPLOY_CLIENTS" ] && [ "$DEPLOY_CLIENTS" != "" ]; then
        # 사용자가 지정한 고객사만 배포
        IFS=',' read -ra CLIENTS_ARRAY <<< "$DEPLOY_CLIENTS"
        CLIENTS=("${CLIENTS_ARRAY[@]}")
        echo "Selected clients: ${CLIENTS[*]}"
      else
        # 모든 고객사 배포 (기본값)
        CLIENTS=("CLIENT1" "CLIENT2")
        echo "Deploying to all clients: ${CLIENTS[*]}"
      fi

      for CLIENT_PREFIX in "${CLIENTS[@]}"; do
        echo ""
        echo "=========================================="
        echo "Processing: $CLIENT_PREFIX"
        echo "=========================================="

        # 환경변수 매핑
        VAR_SUBDOMAIN="${CLIENT_PREFIX}_SUBDOMAIN"
        VAR_TOKEN="${CLIENT_PREFIX}_TOKEN"
        VAR_EMAIL="${CLIENT_PREFIX}_EMAIL"

        eval TARGET_SUBDOMAIN=\$$VAR_SUBDOMAIN
        eval TARGET_TOKEN=\$$VAR_TOKEN
        eval TARGET_EMAIL=\$$VAR_EMAIL

        if [ -z "$TARGET_SUBDOMAIN" ] || [ -z "$TARGET_TOKEN" ]; then
          echo "  Skipping $CLIENT_PREFIX: missing credentials"
          continue
        fi

        echo "Target domain: ${TARGET_SUBDOMAIN}.zendesk.com"
        echo "Target email: ${TARGET_EMAIL:-${COMMON_EMAIL}}"

        # App ID 조회 (환경변수에서)
        VAR_APP_ID="${CLIENT_PREFIX}_APP_ID"
        eval TARGET_APP_ID=\$$VAR_APP_ID

        if [ -z "$TARGET_APP_ID" ]; then
          echo "  Skipping $CLIENT_PREFIX: missing APP_ID"
          continue
        fi

        # zcli 환경변수 인증 설정 (프로필보다 우선순위 높음)
        # 기존 환경변수 unset (다른 고객사 인증 정보 제거)
        unset ZENDESK_SUBDOMAIN ZENDESK_EMAIL ZENDESK_API_TOKEN ZENDESK_TOKEN

        # 현재 고객사 환경변수 설정
        export ZENDESK_SUBDOMAIN="$TARGET_SUBDOMAIN"
        export ZENDESK_EMAIL="${TARGET_EMAIL:-${COMMON_EMAIL}}"
        export ZENDESK_API_TOKEN="$TARGET_TOKEN"

        echo "   Using environment variables for authentication"
        echo "   Subdomain: $ZENDESK_SUBDOMAIN"
        echo "   Email: $ZENDESK_EMAIL"

        # zcli.apps.config.json 파일 생성/업데이트 (app_id + test settings)
        CONFIG_FILE="$APP_DIR_ABS/zcli.apps.config.json"
        MANIFEST_FILE="$APP_DIR_ABS/manifest.json"
        
        # manifest.json에서 parameters 읽어서 test settings 생성 (default 값이 있는 것만)
        # jq로 parameters에서 default가 null이 아니고 빈 문자열이 아닌 것들만 추출
        TEST_SETTINGS_JSON=$(jq -r '.parameters // [] | 
          map(select(.default != null and .default != "")) | 
          reduce .[] as $p ({}; . + {($p.name): $p.default})' "$MANIFEST_FILE" 2>/dev/null || echo "{}")
        
        if [ -f "$CONFIG_FILE" ]; then
          # 기존 파일이 있으면 app_id와 test_settings 업데이트
          jq --argjson app_id "$TARGET_APP_ID" --argjson test_settings "$TEST_SETTINGS_JSON" \
            '.app_id = $app_id | .test_settings = $test_settings' "$CONFIG_FILE" > "$CONFIG_FILE.tmp" && mv "$CONFIG_FILE.tmp" "$CONFIG_FILE"
          echo "Updated zcli.apps.config.json with app_id: ${TARGET_APP_ID} and test_settings"
        else
          # 새로 생성 (app_id + test_settings)
          jq -n --argjson app_id "$TARGET_APP_ID" --argjson test_settings "$TEST_SETTINGS_JSON" \
            '{app_id: $app_id, test_settings: $test_settings}' > "$CONFIG_FILE"
          echo "Created zcli.apps.config.json with app_id: ${TARGET_APP_ID} and test_settings"
        fi

        # zcli apps:update 실행 (재시도 로직 포함)
        echo "Running zcli apps:update..."
        cd "$APP_DIR_ABS"
        
        MAX_RETRIES=3
        RETRY_COUNT=0
        UPDATE_SUCCESS=false
        
        # set -e를 일시적으로 비활성화 (재시도 로직을 위해)
        set +e
        
        while [ $RETRY_COUNT -lt $MAX_RETRIES ]; do
          RETRY_COUNT=$((RETRY_COUNT + 1))
          
          # zcli 실행 및 에러 캡처
          zcli apps:update > /tmp/zcli_update_${CLIENT_PREFIX}.log 2>&1
          ZCLI_EXIT_CODE=$?
          
          if [ $ZCLI_EXIT_CODE -eq 0 ]; then
            UPDATE_SUCCESS=true
            cat /tmp/zcli_update_${CLIENT_PREFIX}.log
            break
          else
            ERROR_LOG=$(cat /tmp/zcli_update_${CLIENT_PREFIX}.log 2>/dev/null || echo "")
            cat /tmp/zcli_update_${CLIENT_PREFIX}.log
            
            # 502 Bad Gateway 에러인 경우에만 재시도
            if echo "$ERROR_LOG" | grep -qE "502|Bad Gateway|ERR_BAD_RESPONSE"; then
              if [ $RETRY_COUNT -lt $MAX_RETRIES ]; then
                echo "[$CLIENT_PREFIX] 502 Bad Gateway error (attempt $RETRY_COUNT/$MAX_RETRIES), retrying in 10 seconds..."
                sleep 10
              else
                echo "[$CLIENT_PREFIX] zcli apps:update FAILED after $MAX_RETRIES attempts (502 errors)"
                set -e
                exit 1
              fi
            else
              echo "[$CLIENT_PREFIX] zcli apps:update FAILED (non-retryable error)"
              set -e
              exit 1
            fi
          fi
        done
        
        # set -e 다시 활성화
        set -e
        
        if [ "$UPDATE_SUCCESS" = "true" ]; then
          echo "[$CLIENT_PREFIX] zcli apps:update SUCCESS"
        fi
        cd "$CI_PROJECT_DIR"
      done

      echo ""
      echo "All clients processed"

  only:
    - main

# ==========================================
# 4. Promote to Prod
# ==========================================
promote_to_prod:
  stage: promote-prod
  needs: ["deploy_to_dev"]
  when: manual
  script:
    - |
      set -e

      TIMESTAMP=$(date +"%Y%m%d-%H%M%S")
      ROLLBACK_PATH="rollback/${TIMESTAMP}"

      echo "Backing up Prod → ${ROLLBACK_PATH}"

      aws s3 sync \
        s3://${PROD_S3_BUCKET} \
        s3://${PROD_S3_BUCKET}/${ROLLBACK_PATH} \
        --exclude "rollback/*"

      echo "Promoting Dev → Prod"

      aws s3 sync \
        s3://${DEV_S3_BUCKET} \
        s3://${PROD_S3_BUCKET} \
        --delete \
        --exclude "rollback/*"

      aws cloudfront create-invalidation \
        --distribution-id ${PROD_CF_ID} \
        --paths "/*"

      echo " Promote to Prod completed"

  only:
    - main

# ==========================================
# 5. Rollback
# ==========================================
rollback_prod:
  stage: rollback
  when: manual
  script:
    - |
      set -e

      LATEST_ROLLBACK=$(aws s3 ls s3://${PROD_S3_BUCKET}/rollback/ | tail -n 1 | awk '{print $2}')

      if [ -z "$LATEST_ROLLBACK" ]; then
        echo " No rollback snapshot found"
        exit 1
      fi

      echo "Rolling back using snapshot: ${LATEST_ROLLBACK}"

      aws s3 sync \
        s3://${PROD_S3_BUCKET}/rollback/${LATEST_ROLLBACK} \
        s3://${PROD_S3_BUCKET} \
        --delete \
        --exclude "rollback/*"

      aws cloudfront create-invalidation \
        --distribution-id ${PROD_CF_ID} \
        --paths "/*"

      echo " Rollback completed"

  only:
    - main

개요

이 CI/CD 파이프라인은 WiseNTalk Zendesk Private App을 여러 고객사 Zendesk 계정에 자동으로 배포하기 위해 구성되었습니다.

주요 특징

  • 자동 빌드: React 앱을 빌드하여 정적 파일 생성
  • Dev 환경 배포: S3 + CloudFront를 통한 개발 환경 배포
  • 선택적 Zendesk 배포: 원하는 고객사만 선택하여 Zendesk 앱 업데이트
  • Prod 승격: 수동 승인 후 프로덕션 환경으로 배포
  • 롤백 기능: 프로덕션 환경 롤백 지원

기술 스택

  • CI/CD: GitLab CI/CD
  • 빌드 도구: npm
  • 배포: AWS S3, CloudFront
  • Zendesk 배포: Zendesk CLI (zcli)
  • 스크립트: Bash

Stages

  1. build: 앱 빌드
  1. deploy-dev: 개발 환경 배포
  1. update-zendesk: Zendesk 앱 업데이트 (Manual)
  1. promote-prod: 프로덕션 승격 (Manual)
  1. rollback: 프로덕션 롤백 (Manual)

3. Update Zendesk Stage

  • 프로필 전환 방식이 아닙니다. → zcli profiles:use 와 같은 명령을 사용하지 않습니다.
  • 환경변수 기반 인증 → zcli는 환경변수가 있으면 프로필보다 우선 사용합니다.
  • 고객사별 순차 처리 :
    • CLIENT1: 환경변수 설정 → zcli apps:update 실행
    • CLIENT2: 환경변수 재설정 → zcli apps:update 실행
위 방식은 프로필 관리가 불필요하며, 환경변수만 재설정하면 즉시 다른 고객사로 전환 가능합니다.

Zendesk 앱 업데이트는 두 가지 방법을 제공합니다.

3-A. 변수 기반 선택적 배포

Job: update_zendesk_dev_apps_zcli

특징:
- DEPLOY_CLIENTS 변수로 배포할 고객사 선택
- 여러 고객사를 순차적으로 배포
- 변수가 없으면 모든 고객사 배포

동작 흐름:
1. zcli 설치 확인bash if ! command -v zcli &> /dev/null; then npm install -g @zendesk/zcli fi

  1. 고객사 목록 결정
    if [ -n "$DEPLOY_CLIENTS" ] && [ "$DEPLOY_CLIENTS" != "" ]; then
  # 사용자가 지정한 고객사만 배포
  IFS=',' read -ra CLIENTS_ARRAY <<< "$DEPLOY_CLIENTS"
  CLIENTS=("${CLIENTS_ARRAY[@]}")
else
  # 모든 고객사 배포 (기본값)
  CLIENTS=("CLIENT1" "CLIENT2")
fi
  1. 각 고객사별 처리 (for loop):
    • 환경변수 매핑 (CLIENT1_SUBDOMAIN, CLIENT1_TOKEN, etc.)
    • zcli 인증 설정 (환경변수 기반)
    • zcli.apps.config.json 생성/업데이트
      • app_id: 고객사별 앱 ID
      • test_settings: manifest.json의 parameters 기본값 자동 추출
    • zcli apps:update 실행 (502 에러 재시도 포함)
  1. 재시도 로직:
    • 최대 3회 재시도
    • 502 Bad Gateway 에러만 재시도
    • 재시도 간격: 10초

  1. CloudFront 캐시 무효화

동작 로직

전체 파이프라인 흐름

1. main 브랜치에 push
   ↓
2. build_app 자동 실행
   - npm install & build
   - build/ 디렉토리 생성
   ↓
3. deploy_to_dev 자동 실행
   - build/ → Dev S3
   - CloudFront 캐시 무효화
   ↓
4. update_zendesk_dev_apps_zcli (Manual)
   - 사용자가 수동 실행
   - DEPLOY_CLIENTS 변수로 고객사 선택
   - zcli로 각 고객사 Zendesk 앱 업데이트
   ↓
5. promote_to_prod (Manual, 독립적)
   - Dev → Prod 승격
   - 자동 백업 생성
   ↓
6. rollback_prod (Manual, 필요시)
   - 최신 스냅샷으로 복원

Zendesk 앱 업데이트 상세 로직

for each CLIENT in CLIENTS:
  1. 환경변수 읽기
     - ${CLIENT}_SUBDOMAIN
     - ${CLIENT}_TOKEN
     - ${CLIENT}_EMAIL (optional)
     - ${CLIENT}_APP_ID

  2. zcli 인증 설정
     export ZENDESK_SUBDOMAIN="..."
     export ZENDESK_EMAIL="..."
     export ZENDESK_API_TOKEN="..."

  3. zcli.apps.config.json 생성
     {
       "app_id": <CLIENT_APP_ID>,
       "test_settings": {
         "vfApplicationName": "WiseN Talk",
         "outboundType": "connect",
         ...
       }
     }
     (test_settings는 manifest.json의 parameters에서 자동 추출)

  4. zcli apps:update 실행
     - 성공 시: 다음 고객사로
     - 502 에러: 재시도 (최대 3회)
     - 기타 에러: 실패 처리

test_settings 자동 생성 로직

manifest.json의 parameters 배열에서:
- default 값이 존재하고
- 빈 문자열이 아닌 항목만 추출
- { "parameter_name": "default_value" } 형태로 변환

예시:

// manifest.json
{
  "parameters": [
    { "name": "vfApplicationName", "default": "WiseN Talk" },
    { "name": "connectInstanceUrl", "default": "" },
    { "name": "defaultCountryPrefix", "default": "+82" }
  ]
}

// → test_settings
{
  "vfApplicationName": "WiseN Talk",
  "defaultCountryPrefix": "+82"
}

사용 방법

1. 일반적인 배포 워크플로우

Step 1: 코드 변경 및 커밋

git add .
git commit -m "feat: 새로운 기능 추가"
git push origin main

Step 2: 자동 빌드 및 Dev 배포

  • main 브랜치에 push하면 자동으로 실행됨
  • GitLab CI/CD → Pipelines에서 진행 상황 확인

Step 3: Zendesk 앱 업데이트 (Manual)

방법 A: 변수 기반 선택적 배포
1. GitLab CI/CD → Pipelines → update_zendesk_dev_apps_zcli 클릭
2. “Run pipeline” 클릭
3. Variables 섹션에 DEPLOY_CLIENTS 입력:
- CLIENT1 → CLIENT1만 배포
- CLIENT2 → CLIENT2만 배포
- CLIENT1,CLIENT2 → 둘 다 배포
- 비워두기 → 모든 고객사 배포
4. “Run job” 클릭

방법 B: 개별 Job 사용
1. GitLab CI/CD → Pipelines에서 원하는 job 선택:
- update_zendesk_client1 → CLIENT1만
- update_zendesk_client2 → CLIENT2만
2. “Run job” 클릭

Step 4: 프로덕션 승격 (Manual, 독립적)

  1. Dev 환경에서 테스트 완료 후
  1. promote_to_prod job 실행
  1. 자동으로 백업 생성 후 Prod에 배포

2. 특정 고객사만 배포하는 경우

시나리오: CLIENT1만 업데이트하고 싶을 때

방법 1: update_zendesk_client1 job 실행

방법 2: update_zendesk_dev_apps_zcli 실행 시 Variables에 DEPLOY_CLIENTS=CLIENT1 입력

3. 롤백 방법

프로덕션에 문제가 발생한 경우:

  1. GitLab CI/CD → Pipelines
  1. rollback_prod job 실행
  1. 최신 스냅샷으로 자동 복원
