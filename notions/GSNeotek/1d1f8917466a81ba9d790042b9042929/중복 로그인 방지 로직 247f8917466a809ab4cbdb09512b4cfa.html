1. LocalStorage + Timestamp 방식

| 비용: 무료

// 간단한 로컬 기반 체크
function checkDuplicateLogin() {
    const key = `softphone_session_${userId}`;
    const now = Date.now();
    const existing = localStorage.getItem(key);

    if (existing) {
        const data = JSON.parse(existing);
        if (now - data.timestamp < 30000) { // 30초 이내
            return false; // 중복 로그인
        }
    }

    localStorage.setItem(key, JSON.stringify({
        sessionId: generateSessionId(),
        timestamp: now
    }));

    return true; // 로그인 허용
}

✅ 장점:

  • 구현이 매우 간단
  • 외부 의존성 없음
  • 즉시 적용 가능

❌ 단점:

  • 브라우저 간 동기화 불가 (Chrome ↔ Edge)
  • 시간 기반이라 부정확할 수 있음

// shared-worker.js
const sessions = new Set();
const ports = [];

self.addEventListener('connect', (event) => {
    const port = event.ports[0];
    ports.push(port);

    port.onmessage = (e) => {
        if (e.data.type === 'REGISTER') {
            if (sessions.size === 0) {
                sessions.add(e.data.sessionId);
                port.postMessage({ type: 'LEADER', sessionId: e.data.sessionId });

                port.postMessage({ type: 'FOLLOWER' });

// WebSocket 서버 (Node.js)
const WebSocket = require('ws');
const wss = new WebSocket.Server({ port: 8080 });

const activeSessions = new Map();

wss.on('connection', (ws) => {
    ws.on('message', (message) => {
        const data = JSON.parse(message);

        if (data.type === 'REGISTER') {
            const existing = activeSessions.get(data.userId);

            if (!existing) {
                activeSessions.set(data.userId, ws);
                ws.send(JSON.stringify({ type: 'LEADER' }));
            } else {
                ws.send(JSON.stringify({ type: 'FOLLOWER' }));
            }
        }
    });
});

| 비용: 중비용 ($100/월)

// Redis 기반 세션 관리
const redis = require('redis');
const client = redis.createClient();

class SessionManager {
    async registerSession(userId, sessionId) {
        const key = `session:${userId}`;
        const existing = await client.hgetall(key);

        if (Object.keys(existing).length === 0) {
            // 새로운 세션
            await client.hset(key, {
                sessionId,
                timestamp: Date.now(),
                heartbeat: Date.now()
            });
            await client.expire(key, 30); // 30초 TTL
            return { isLeader: true };
        } else {
            // 기존 세션 존재
            return { isLeader: false, existingSession: existing.sessionId };
        }
    }

    async heartbeat(userId, sessionId) {
        const key = `session:${userId}`;
        const session = await client.hgetall(key);

        if (session.sessionId === sessionId) {
            await client.hset(key, 'heartbeat', Date.now());
            await client.expire(key, 30);
            return { success: true };
        }
        return { success: false };
    }
}

class AWSSessionManager {
    async registerSession(userId, sessionData) {
        const key = `session:${userId}`;
        const ttl = 30; // 30초

        try {
            // SET IF NOT EXISTS with TTL
            const result = await client.set(key, JSON.stringify(sessionData), 'EX', ttl, 'NX');

            if (result === 'OK') {
                return { isLeader: true };
            } else {
                const existing = await client.get(key);
                return { isLeader: false, existingSession: JSON.parse(existing) };
            }
        } catch (error) {
            console.error('Session registration failed:', error);
            return { isLeader: false, error: error.message };
        }
    }
}

10. Amazon DynamoDB

| 비용: 중비용 ($50-100/월)

// DynamoDB 기반 세션 관리
const AWS = require('aws-sdk');
const dynamodb = new AWS.DynamoDB.DocumentClient();

class DynamoDBSessionManager {
    async registerSession(userId, sessionData) {
        const params = {
            TableName: 'UserSessions',
            Item: {
                userId: userId,
                sessionId: sessionData.sessionId,
                timestamp: Date.now(),
                ttl: Math.floor(Date.now() / 1000) + 30 // 30초 후 자동 삭제
            },
            ConditionExpression: 'attribute_not_exists(userId)' // 존재하지 않을 때만 생성
        };

        try {
            await dynamodb.put(params).promise();
            return { isLeader: true };
        } catch (error) {
            if (error.code === 'ConditionalCheckFailedException') {
                // 이미 세션 존재
                const existing = await this.getSession(userId);
                return { isLeader: false, existingSession: existing };
            }
            throw error;
        }
    }

    async heartbeat(userId, sessionId) {
        const params = {
            TableName: 'UserSessions',
            Key: { userId },
            UpdateExpression: 'SET #ttl = :ttl, #heartbeat = :heartbeat',
            ConditionExpression: 'sessionId = :sessionId',
            ExpressionAttributeNames: {
                '#ttl': 'ttl',
                '#heartbeat': 'lastHeartbeat'
            },
            ExpressionAttributeValues: {
                ':ttl': Math.floor(Date.now() / 1000) + 30,
                ':heartbeat': Date.now(),
                ':sessionId': sessionId
            }
        };

        try {
            await dynamodb.update(params).promise();
            return { success: true };
        } catch (error) {
            return { success: false, error: error.message };
        }
    }
}

✅ 장점:

  • 자동 확장
  • TTL 기능으로 자동 정리
  • 99.99% 가용성

❌ 단점:

  • Eventually Consistent 특성
  • AWS 종속성

class HybridSessionManager {
    constructor() {
        this.layers = [
            new BroadcastChannelLayer(),    // Layer 1: 같은 브라우저
            new LocalStorageLayer(),        // Layer 2: 로컬 백업
            new ServerAPILayer()            // Layer 3: 글로벌 동기화
        ];
    }

    async checkDuplicateLogin() {
        // 1단계: 빠른 로컬 체크
        const localResult = await this.layers[0].check();
        if (!localResult.isLeader) {
            return localResult;
        }

        // 2단계: 브라우저 간 체크
        const globalResult = await this.layers[2].check();
        return globalResult;
    }
}

    } catch (error) {
        if (error.code === 'ConditionalCheckFailedException') {
            const existing = await getSession(userId);
            return createResponse(200, {
                isLeader: false,
                existingSession: existing
            });
        }
        throw error;
    }
}

function createResponse(statusCode, body) {
    return {
        statusCode,
        headers: {
            'Access-Control-Allow-Origin': '*',
            'Access-Control-Allow-Methods': 'GET,POST,PUT,DELETE,OPTIONS',
            'Access-Control-Allow-Headers': 'Content-Type,Authorization',
            'Content-Type': 'application/json'
        },
        body: JSON.stringify(body)
    };
}

// elasticache-session-manager.js
const redis = require('redis');

class ElastiCacheSessionManager {
    constructor(redisEndpoint) {
        this.client = redis.createClient({
            host: redisEndpoint,
            port: 6379,
            retry_strategy: (options) => {
                if (options.error && options.error.code === 'ECONNREFUSED') {
                    return new Error('Redis server connection refused');
                }
                if (options.total_retry_time > 1000 * 60 * 60) {
                    return new Error('Retry time exhausted');
                }
                return Math.min(options.attempt * 100, 3000);
            }
        });
    }

    async registerSession(userId, sessionData) {
        const key = `session:${userId}`;
        const sessionJson = JSON.stringify({
            sessionId: sessionData.sessionId,
            timestamp: Date.now(),
            browserInfo: sessionData.browserInfo
        });

        // SET with NX (Not eXists) and EX (EXpire)
        const result = await this.client.set(key, sessionJson, 'EX', 30, 'NX');

        if (result === 'OK') {
            return { isLeader: true, sessionId: sessionData.sessionId };
        } else {
            const existing = await this.client.get(key);
            return { isLeader: false, existingSession: JSON.parse(existing) };
        }
    }

    async heartbeat(userId, sessionId) {
        const key = `session:${userId}`;
        const current = await this.client.get(key);

        if (current) {
            const session = JSON.parse(current);
            if (session.sessionId === sessionId) {
                session.lastHeartbeat = Date.now();
                await this.client.setex(key, 30, JSON.stringify(session));
                return { success: true };
            }
        }

        return { success: false, message: 'Session not found or invalid' };
    }

    async cleanup(userId, sessionId) {
        const key = `session:${userId}`;
        const current = await this.client.get(key);

        if (current) {
            const session = JSON.parse(current);
            if (session.sessionId === sessionId) {
                await this.client.del(key);
                return { success: true };
            }
        }

        return { success: false };
    }

    // 모니터링용 메서드
    async getActiveSessionCount() {
        const keys = await this.client.keys('session:*');
        return keys.length;
    }

    async getAllActiveSessions() {
        const keys = await this.client.keys('session:*');
        const sessions = [];

        for (const key of keys) {
            const session = await this.client.get(key);
            if (session) {
                sessions.push({
                    userId: key.replace('session:', ''),
                    ...JSON.parse(session)
                });
            }
        }

        return sessions;
    }
}

class SecureSessionManager {

        return jwt.sign(
            {
                userId,
                sessionId,
                timestamp: Date.now()
            },
            process.env.JWT_SECRET,
            {
                expiresIn: '30m',
                issuer: 'softphone-session-manager',
                audience: 'softphone-clients'
            }
        );
    }

    validateSessionToken(token) {
        try {
            const decoded = jwt.verify(token, process.env.JWT_SECRET);
            return { valid: true, payload: decoded };
        } catch (error) {
            return { valid: false, error: error.message };
        }
    }

    async registerSecureSession(userId, sessionData, authToken) {
        // 1. 토큰 검증
        const tokenValidation = this.validateSessionToken(authToken);
        if (!tokenValidation.valid) {
            throw new Error('Invalid authentication token');
        }

        // 2. 사용자 권한 확인
        if (tokenValidation.payload.userId !== userId) {
            throw new Error('User ID mismatch');
        }

        // 3. 세션 등록
        return await this.registerSession(userId, sessionData);
    }
}

중복 티켓 발행 트러블슈팅 가이드

제공해주신 AWS.js, CCP.js, Zendesk.js 코드를 기반으로 Zendesk 중복 티켓 발행 문제에 대한 트러블슈팅 가이드를 제공해 드립니다.

1. 문제의 핵심 원인: 티켓 생성 로직의 중복 호출

분석 결과, 중복 티켓 발행의 가장 큰 원인은 하나의 전화 인입에 대해 티켓 생성 로직이 여러 번 호출될 가능성이 있기 때문입니다. 특히 CCP.js의 Amazon Connect 이벤트 핸들러와 Zendesk.js의 티켓 생성 함수 사이의 상호작용에서 문제가 발생할 수 있습니다.

주요 원인은 다음과 같습니다.

  • handleContactConnecting과 handleContactConnected의 경쟁 상태 (Race Condition):
    • 전화가 연결되는 과정에서 handleContactConnecting (연결 중) 이벤트와 handleContactConnected (연결 완료) 이벤트가 거의 동시에 발생할 수 있습니다.
    • 두 이벤트 핸들러 모두 Zendesk.js의 searchUserByPhoneNumber 함수를 호출하며, 이 함수는 특정 조건에서 createTicketByVoice를 호출하여 티켓을 생성합니다.
    • 만약 두 이벤트가 모두 티켓 생성 조건에 부합하게 되면, 같은 전화에 대해 두 개의 티켓이 생성될 수 있습니다.
  • handleContactConnected의 중복 실행 가능성:
    • CCP.js의 handleContactConnected 함수 내에 lastContactId를 사용하여 중복 실행을 방지하려는 시도가 있습니다.JavaScript

      if (lastContactId === contact_id) {
      lastContactId = contact_id;
      return;
      } else {
      lastContactId = contact_id;
      }

    • 하지만 이 로직은 비동기 환경에서는 완벽하지 않을 수 있으며, 특정 상황(예: 빠른 재연결, 네트워크 지연)에서는 중복 호출을 막지 못할 수 있습니다.
  • 티켓 존재 여부 확인 로직 부재:
    • Zendesk.js의 createTicketByVoice 함수가 호출될 때, 해당 contactId에 대한 티켓이 이미 존재하는지 확인하는 절차가 없습니다. 따라서 로직이 중복 호출되면 티켓도 중복으로 생성됩니다.

2. 해결 방안

아래의 단계에 따라 코드를 수정하면 중복 티켓 발행 문제를 해결할 수 있습니다.

1단계: 티켓 생성 시점 단일화

가장 중요한 것은 티켓 생성 로직이 오직 한 번만, 그리고 가장 확실한 시점에 호출되도록 보장하는 것입니다.

  • 권장 사항: 티켓 생성은 전화 연결이 완전히 확인된 handleContactConnected 이벤트에서만 수행하도록 로직을 수정합니다. handleContactConnecting에서는 고객 정보 조회 및 화면 팝업 등 사전 작업만 처리하도록 역할을 분리합니다.
  • 수정 가이드:
    1. CCP.js의 handleContactConnecting 함수에서 zendeskFunc를 호출하는 부분은 그대로 두되, Zendesk.js의 searchUserByPhoneNumber 함수를 수정합니다.
    1. Zendesk.js의 searchUserByPhoneNumber 함수 내에서 eventName이 'connecting'일 경우에는 절대로 createTicketByVoice가 호출되지 않도록 수정합니다.

2단계: 티켓 생성 전 중복 확인 로직 추가

티켓을 생성하기 직전에, 동일한 contactId로 생성된 티켓이 이미 있는지 Zendesk API를 통해 확인하는 절차를 추가해야 합니다.

  • 권장 사항: Zendesk.js에 contactId를 기반으로 티켓을 검색하는 함수를 추가하고, createTicketByVoice 호출 전에 이 함수를 실행하여 중복을 방지합니다. Zendesk 티켓의 external_id 필드를 contactId로 활용하면 효율적인 조회가 가능합니다.
  • 구현 예시 (Zendesk.js):JavaScript

    // 티켓 생성 전, external_id를 사용해 이미 티켓이 있는지 확인하는 함수
    findTicketByContactId = async (contactId) => {
    const tickets = await this.getFromZD(`search.json?query=type:ticket+external_id:${contactId}`, 'results', []);

    }

    }

    // searchUserByPhoneNumber 함수 내에서 티켓 생성 로직 수정
    // ... (기존 로직)
    if (eventName === "connected" && callInfo.type === "INBOUND") {
    const existingTicket = await this.findTicketByContactId(callInfo.contactId);
    if (existingTicket) {
    console.log(`Ticket already exists for contactId ${callInfo.contactId}: Ticket ID ${existingTicket.id}`);
    // 이미 티켓이 존재하므로 추가 생성하지 않고, 필요하다면 기존 티켓을 업데이트하거나 화면에 표시
    this.openTicket(assignee_id, existingTicket.id);
    } else {
    // 티켓이 없을 때만 생성
    ticketId = await this.createTicketByVoice(callInfo);
    this.openTicket(assignee_id, ticketId);
    // ... (이후 로직)
    }
    }

3단계: 강력한 중복 실행 방지 (Debounce) 구현

lastContactId 변수를 사용하는 현재 방식보다 더 안정적인 중복 실행 방지 메커니즘을 도입합니다.

  • 권장 사항: Map 객체를 사용하여 각 contactId의 처리 상태(processing, done)를 관리합니다.
  • 구현 예시 (CCP.js):JavaScript

    // 파일 상단에 Map 객체 선언
    const contactProcessingState = new Map();

    // handleContactConnected 함수 수정
    handleContactConnected = async (contact) => {
    try {
    const contact_id = contact.getContactId();

    if (contactProcessingState.get(contact_id) === 'processing' || contactProcessingState.get(contact_id) === 'done') {
    console.log(`Contact ${contact_id} is already being processed or has been processed.`);
    return;
    }
    contactProcessingState.set(contact_id, 'processing');

    // ... 기존 handleContactConnected 로직 ...

    // 모든 로직이 성공적으로 완료된 후 상태 변경
    contactProcessingState.set(contact_id, 'done');

    } catch (e) {
    console.log("[handleContactConnected] ERR", e);
    const contact_id = contact ? contact.getContactId() : 'unknown';
    contactProcessingState.set(contact_id, 'error'); // 에러 발생 시 상태 기록
    }
    }

3. 추가 조치: 상세 로그 추가

문제 해결 과정을 명확히 추적하기 위해 주요 로직 실행 시점에 상세한 로그를 추가하는 것이 좋습니다.

  • CCP.js의 handleContactConnecting, handleContactConnected 이벤트 발생 시점
  • Zendesk.js의 searchUserByPhoneNumber, createTicketByVoice, findTicketByContactId 함수 호출 및 반환 시점
  • 각 로그에는 contactId를 반드시 포함하여 특정 통화의 흐름을 추적할 수 있도록 해야 합니다.

위의 3단계 해결 방안을 적용하면 Zendesk 티켓이 중복으로 생성되는 문제를 근본적으로 해결할 수 있을 것입니다.

네, 좋은 질문입니다. 제가 왜 두 이벤트 핸들러(handleContactConnecting, handleContactConnected)가 모두 Zendesk.js의 searchUserByPhoneNumber 함수를 호출한다고 말씀드렸는지 코드상의 근거를 구체적으로 설명해 드리겠습니다.

핵심은 CCP.js 컴포넌트가 Zendesk.js 컴포넌트의 함수를 this.props.zendeskFunc라는 props를 통해 호출하는 구조에 있습니다.

근거 1: handleContactConnecting에서의 호출

CCP.js 파일의 handleContactConnecting 함수를 보면, 전화가 연결되기 시작하는 시점에 아래와 같은 코드가 있습니다.

JavaScript

// CCP.js
handleContactConnecting = async (contact) => {
// ... (중략) ...
if (contact.getType() === 'voice') {
// ... (중략) ...
let monitoringF = isSilentMonitor || isBarge;
if (this.state.zendesk_use_f === 'Y' && !monitoringF) {
this.props.zendeskFunc("searchUserByPhoneNumber", callInfo, "connecting");
}
// ... (후략) ...
}

  • 이 코드 블록은 전화(voice)가 들어오고, Zendesk 연동이 활성화(zendesk_use_f === 'Y') 되어 있으며, 모니터링콜이 아닐 경우 this.props.zendeskFunc를 호출합니다.
  • 첫 번째 인자로 "searchUserByPhoneNumber"라는 문자열을, 세 번째 인자로 "connecting"을 전달하여 Zendesk.js에 정의된 searchUserByPhoneNumber 함수를 실행하도록 요청하고 있습니다.

근거 2: handleContactConnected에서의 호출

마찬가지로 CCP.js 파일의 handleContactConnected 함수를 보면, 전화 연결이 완료된 시점에도 zendeskFunc를 호출하는 부분이 있습니다.

JavaScript

// CCP.js
handleContactConnected = async (contact) => {
try {
// ... (중략) ...
if (contact.getType() === 'voice') {
// ... (중략) ...
if (this.state.zendesk_use_f === "Y" && !monitoringF) {
if (this.props.callInfo.refreshF === false) {
let callInfo = this.props.callInfo;
// ... (중략) ...
if (call_type === "INBOUND") {
// ... (중략) ...
this.props.updateCallInfo(callInfo);
await this.props.zendeskFunc("searchUserByPhoneNumber", callInfo, "connected");
}
// ... (후략) ...
}
}
// ... (후략) ...
}
} catch (e) {
// ...
}
}

  • 이 코드 블록은 전화가 완전히 연결된(connected) 후, Zendesk 연동이 활성화되어 있고, 새로고침된 상황이 아닐 때 this.props.zendeskFunc를 호출합니다.
  • 여기서도 마찬가지로 첫 번째 인자로 "searchUserByPhoneNumber"를 전달하여 동일한 함수를 다시 호출하고 있습니다. 다만 이번에는 세 번째 인자로 "connected"를 전달하여 티켓 생성과 같은 후속 조치를 하도록 유도합니다.

결론

위 코드에서 보시다시피, CCP.js는 connecting 상태와 connected 상태, 즉 두 번의 다른 시점에서 동일한 searchUserByPhoneNumber 함수를 호출하고 있습니다. 이 구조가 각 단계에서 티켓 생성 조건이 충족될 경우, 중복 티켓이 발행될 수 있는 직접적인 원인이 되는 것입니다.
